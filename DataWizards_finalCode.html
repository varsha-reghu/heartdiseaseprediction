from sklearn.pipeline import Pipeline
from sklearn.ensemble import RandomForestClassifier
from sklearn.svm import SVC
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import GridSearchCV
from tqdm import tqdm

model_names = ['Random Forest', 'SVM', 'Logsitic Regression']
# Create a pipeline for Random Forest
rf_pipeline = Pipeline([
    ('clf', RandomForestClassifier())
])

# Define parameters for Random Forest
rf_params = {
    'clf__n_estimators': [50,100, 200],
    'clf__max_depth': [3, 5, 10],
    'clf__min_samples_split': [2, 5]
}

# Create a pipeline for SVM
svm_pipeline = Pipeline([
    ('clf', SVC())
])

# Define parameters for SVM
svm_params = {
    'clf__C': [0.1, 1],
    'clf__kernel': ['linear', 'rbf']
}

# Create a pipeline for Logistic Regression
lr_pipeline = Pipeline([
    ('clf', LogisticRegression())
])

# Define parameters for Logistic Regression
lr_params = {
    'clf__C': [0.1, 1],
    'clf__tol': [.0001,.001, .01]
}

# List of pipelines and parameter grids
pipelines = [rf_pipeline, svm_pipeline, lr_pipeline]
params = [rf_params, svm_params, lr_params]

# Fit the pipelines using GridSearchCV
best_estimators = []
for pipeline, param_grid in tqdm(zip(pipelines, params)):
    grid_search = GridSearchCV(pipeline, param_grid, cv=5, n_jobs=-1)
    grid_search.fit(X_train, y_train)
    best_estimators.append(grid_search.best_estimator_)

for estimator, name in zip(best_estimators, ['Random Forest', 'SVM', 'Logistic Regression']):
    print('-'*20)
    print('Model:',name)
    print('Best Parameters:',estimator[0])
    
    # Predict on the test set
    y_pred = estimator.predict(X_test)

    # Print classification report
    print(classification_report(y_test, y_pred))
